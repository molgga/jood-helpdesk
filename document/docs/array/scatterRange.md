# scatterRange

source 배열의 분포도 정보를 반환합니다.

```ts
const result = scatterRange({
  source: [130, 128, 131, 134, 142, 150, 149, 147, 143],
  min: 100,
  max: 200,
  split: 11
});
```

```ts
// result
{
  min: 100,
  max: 200,
  axis: [
    100, 110, 120, 130,
    140, 150, 160, 170,
    180, 190, 200
  ],

    [ 100, 110 ], [ 110, 120 ],
    [ 120, 130 ], [ 130, 140 ],
    [ 140, 150 ], [ 150, 160 ],
    [ 160, 170 ], [ 170, 180 ],
    [ 180, 190 ], [ 190, 200 ],
    [ 200, 400 ]
  ],
  series: [
    [ 100, 0 ], [ 110, 0 ],
    [ 120, 1 ], [ 130, 3 ],
    [ 140, 4 ], [ 150, 1 ],
    [ 160, 0 ], [ 170, 0 ],
    [ 180, 0 ], [ 190, 0 ],

}
```
